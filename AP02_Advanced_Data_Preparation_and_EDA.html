# Extract numerical values (note the 'join' function)
price = []
for i in df_orig['price_raw']:
    d1 = re.findall(  "CHF\s*(\d+)\.-", i)
    try:
        d2 = int(''.join(d1))
    except:
        d2 = None
    price.append(d2)

# Save as new variable in the pandas data frame
df_orig['price'] = pd.Series(price, dtype="Int64")
